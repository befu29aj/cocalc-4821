\aufgabe{5} Use Cramer's rule to find the values of $x_{2}$ and $x_{3}$.\begin{equation}
\begin{array}{ccccccc}
x_{1}&+&x_{2}&+&2x_{3}&=&-1\\
2x_{1}&-&x_{2}&+&2x_{3}&=&-4\\
4x_{1}&+&x_{2}&+&4x_{3}&=&-2
\end{array}
\end{equation}

\aufgabe{5} Find the product $AB$.
\begin{equation}
A=\left[\begin{array}{cc}
3+3i&6i\\
-5&-4-7i
\end{array}\right]\hspace{.5in}
B=\left[\begin{array}{cc}
-5+6i&-4-7i\\
2-5i&-7+2i
\end{array}\right]
\end{equation}

\aufgabe{5} Find the determinant of $C$.

C=\left[\begin{array}{cccc}
-2&1&-3&0\\
0&1&3&0\\
-1&0&0&3\\
1&2&0&2
\end{array}\right]
\end{equation}
